#include <stdio.h>
int nn;
int A[256];

int power(int w)
{
	int i,r;
	r=1;
	for(i=1;i<=w;i++)
		r=r*2;
	return r;
}

void next(int m)
{
	int i;
	if(m<nn)
	{
		for(i=0;i<=1;i++)
		{
			A[m]=i;
			next(m+1);
		}
	}
	else
		if(check(nn-1))
			out();
}
